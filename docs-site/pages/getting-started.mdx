# Quick Start

Build your first Permaweb OS application in 5 minutes.

## Prerequisites

- A wallet (Arweave or Ethereum)
- Node.js 18+
- A deployed Permaweb OS cluster (or [set one up](/cluster-setup))

## 1. Install the SDK

```bash
npm install @permaweb/sdk
```

Or use the minimal fetch-based approach (no SDK needed):

```typescript
// No SDK required - just fetch and crypto
import { signatureHeaders } from 'http-message-sig';
```

## 2. Connect Your Wallet

Permaweb OS supports both Arweave and Ethereum wallets. The system auto-detects your wallet type from the address format.

### Arweave Wallet

Arweave addresses are Base64URL-encoded 43-character strings (e.g., `aNBU7-1-2-3-4...`).

```typescript
// Arweave wallet (browser extension)
const wallet = await window.arweaveWallet.connect(['SIGN']);

// Or using Arweave JS
import Arweave from 'arweave';
const arweave = Arweave.init({});
const wallet = await arweave.wallets.jwkToAddress(keyfile);
```

### Ethereum Wallet

Ethereum addresses are 42-character hex strings starting with `0x` (e.g., `0x1234...`).

```typescript
// Ethereum wallet (MetaMask)
const wallet = await window.ethereum.request({ method: 'eth_requestAccounts' });

// Or using ethers.js
import { Wallet } from 'ethers';
const wallet = Wallet.createRandom();
```

### Wallet Type Detection

The system automatically detects wallet type from the address:

| Wallet Type | Address Format | Example |
|------------|----------------|---------|
| Arweave | Base64URL, 43 chars | `aNBU7t...` (no `0x` prefix) |
| Ethereum | Hex, 42 chars with `0x` | `0xABC123...` |

## 3. Create Your Pod

```typescript
// Request a nonce from the gateway
const nonceResponse = await fetch('https://api.permaweb.live/api/auth/nonce', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ address: wallet.address })
});

const { nonce } = await nonceResponse.json();

// Sign the nonce (method varies by wallet type)
const signature = await wallet.signMessage(nonce);

// Verify and get session token
const verifyResponse = await fetch('https://api.permaweb.live/api/auth/verify', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    walletAddress: wallet.address,
    signature,
    nonce
  })
});

const { sessionToken, pod } = await verifyResponse.json();

// Pod response includes your subdomain
console.log(`Your pod: ${pod.subdomain}`); // e.g., abc123.permaweb.live
```

## 4. Send a Signed Request

Every request to your pod must be signed with HTTP Message Signatures (RFC 9421).

### Ethereum Signing

```typescript
import { signatureHeaders } from 'http-message-sig';

async function sendMessage(content: string, podSubdomain: string) {
  const request = {
    id: `req_${Date.now()}`,
    method: 'message',
    params: { content }
  };

  // Create HTTPSig signature (Ethereum personal_sign)
  const headers = await signatureHeaders({
    method: 'POST',
    url: '/verify',
    headers: {
      host: podSubdomain,
      date: new Date().toUTCString()
    }
  }, {
    signer: {
      keyid: wallet.address.toLowerCase(),
      sign: async (data) => {
        const sig = await wallet.signMessage(data);
        return Buffer.from(sig.slice(2), 'hex');
      }
    }
  });

  // Send to your pod's subdomain
  const response = await fetch(`https://${podSubdomain}/verify`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Signature': headers.Signature,
      'Signature-Input': headers['Signature-Input'],
      'Date': headers.date
    },
    body: JSON.stringify(request)
  });

  return response.body;
}
```

### Arweave Signing

Arweave uses RSA-PSS-SHA256 signatures instead of Ethereum's personal_sign:

```typescript
import { signatureHeaders } from 'http-message-sig';

async function sendMessageArweave(content: string, podSubdomain: string) {
  const request = {
    id: `req_${Date.now()}`,
    method: 'message',
    params: { content }
  };

  // Arweave uses RSA-PSS-SHA256 algorithm
  const headers = await signatureHeaders({
    method: 'POST',
    url: '/verify',
    headers: {
      host: podSubdomain,
      date: new Date().toUTCString()
    }
  }, {
    signer: {
      keyid: walletAddress, // Arweave address (43 chars)
      alg: 'arweave-rsa-pss-sha256',
      sign: async (data) => {
        // Arweave wallet extension signing
        const signature = await window.arweaveWallet.signMessage(
          new TextEncoder().encode(data),
          { name: 'RSA-PSS', saltLength: 32 }
        );
        return new Uint8Array(signature);
      }
    }
  });

  const response = await fetch(`https://${podSubdomain}/verify`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Signature': headers.Signature,
      'Signature-Input': headers['Signature-Input'],
      'Date': headers.date
    },
    body: JSON.stringify(request)
  });

  return response.body;
}
```

## 5. Stream the Response

```typescript
// Parse JSONL stream
async function streamResponse(responseBody: ReadableStream<Uint8Array>) {
  const reader = responseBody.getReader();
  const decoder = new TextDecoder();

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;

    const chunk = decoder.decode(value);
    for (const line of chunk.split('\n')) {
      if (line.trim()) {
        const event = JSON.parse(line);
        
        switch (event.type) {
          case 'text':
            process.stdout.write(event.text);
            break;
          case 'tool_use':
            console.log(`[Tool: ${event.tool}]`, event.args);
            break;
          case 'tool_result':
            console.log(`[Result]`, event.result);
            break;
          case 'done':
            console.log('\n[Complete]');
            break;
          case 'error':
            console.error(`[Error: ${event.code}]`, event.message);
            break;
        }
      }
    }
  }
}

// Usage
const stream = await sendMessage('Create a hello.ts file', 'abc123.permaweb.live');
await streamResponse(stream);
```

## Complete Example

```typescript
// app.ts - Minimal Permaweb OS client (Ethereum)
import { signatureHeaders } from 'http-message-sig';

async function main() {
  // 1. Connect wallet
  const { Wallet } = await import('ethers');
  const wallet = Wallet.createRandom();
  console.log('Wallet address:', wallet.address);

  // 2. Get nonce
  const nonceRes = await fetch('https://api.permaweb.live/api/auth/nonce', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ address: wallet.address })
  });
  const { nonce } = await nonceRes.json();

  // 3. Sign and verify
  const signature = await wallet.signMessage(nonce);
  const verifyRes = await fetch('https://api.permaweb.live/api/auth/verify', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ walletAddress: wallet.address, signature, nonce })
  });
  const { pod } = await verifyRes.json();
  console.log('Pod created:', pod.subdomain);

  // 4. Send signed message
  const request = { id: 'req_1', method: 'message', params: { content: 'Hello!' } };
  const message = JSON.stringify(request);
  
  const sigHeaders = await signatureHeaders({
    method: 'POST',
    url: '/verify',
    headers: { host: pod.subdomain }
  }, {
    signer: {
      keyid: wallet.address,
      sign: async (data) => Buffer.from((await wallet.signMessage(data)).slice(2), 'hex')
    }
  });

  const res = await fetch(`https://${pod.subdomain}/verify`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Signature': sigHeaders.Signature,
      'Signature-Input': sigHeaders['Signature-Input']
    },
    body: message
  });

  // 5. Stream response
  for await (const line of res.body!) {
    const event = JSON.parse(new TextDecoder().decode(line));
    console.log(event);
  }
}

main().catch(console.error);
```
